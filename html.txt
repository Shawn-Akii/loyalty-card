        // 顯示點數圖示
        function displayPointIcons(customer) {
            const pointIconsBlock = document.getElementById('pointIconsBlock');
            pointIconsBlock.innerHTML = '';
            
            // 計算需要多少行來顯示所有點數
            const iconsPerRow = 10;
            const totalRows = Math.ceil(customer.totalPoints / iconsPerRow);
            
            let iconIndex = 0;
            
            for (let row = 0; row < totalRows; row++) {
                const rowDiv = document.createElement('div');
                rowDiv.className = 'point-icons-row';
                
                // 計算這一行需要顯示多少個點數
                const iconsInThisRow = Math.min(iconsPerRow, customer.totalPoints - iconIndex);
                
                for (let i = 0; i < iconsInThisRow; i++) {
                    const pointIcon = document.createElement('div');
                    if (iconIndex < customer.usedPoints) {
                        pointIcon.className = 'point-icon used-point';
                        pointIcon.innerHTML = '<i class="fas fa-check"></i>';
                    } else {
                        pointIcon.className = 'point-icon unused-point';
                        pointIcon.innerHTML = '<i class="fas fa-star"></i>';
                    }
                    rowDiv.appendChild(pointIcon);
                    iconIndex++;
                }
                
                pointIconsBlock.appendChild(rowDiv);
            }
        }

        // 使用點數
        function usePoint(phone, service) {
            const customer = customers.find(c => c.phone === phone);
            
            if (customer) {
                if (customer.usedPoints < customer.totalPoints) {
                    customer.usedPoints += 1;
                    
                    // 創建服務歷史記錄
                    const now = new Date();
                    const timeString = `${now.getFullYear()}-${String(now.getMonth() + 1).padStart(2, '0')}-${String(now.getDate()).padStart(2, '0')} ${String(now.getHours()).padStart(2, '0')}:${String(now.getMinutes()).padStart(2, '0')}`;
                    
                    const historyRecord = {
                        phone: phone,
                        service: service,
                        time: timeString,
                        usageCount: customer.usedPoints
                    };
                    
                    serviceHistory.push(historyRecord);
                    
                    // 更新顯示
                    displayCustomerInfo(customer);
                    displayServiceHistory(phone);
                    
                    // 保存數據
                    saveData();
                    
                    showNotification(`已使用點數，服務：${service}`, 'success');
                } else {
                    showNotification('點數已用完，無法再使用', 'error');
                }
            }
        }

                if (name && phone && points >= 10 && points <= 20) {
                    // 檢查手機號碼是否已存在
                    const existingCustomer = customers.find(c => c.phone === phone);
                    if (existingCustomer) {
                        showNotification('此手機號碼已存在', 'error');
                        return;
                    }
                    
                    // 新增客戶
                    const newCustomer = {
                        name: name,
                        phone: phone,
                        totalPoints: points,
                        usedPoints: 0
                    };
                    
                    customers.push(newCustomer);
                    saveData();
                    
                    // 重置表單
                    addCustomerForm.reset();
                    
                    showNotification(`已新增客戶：${name}`, 'success');
                } else {
                    if (points < 10 || points > 20) {
                        pointsError.classList.remove('hidden');
                    }
                    showNotification('請填寫所有必填欄位', 'error');
                }
            });
        }

        // 匯入資料
        function importData(file) {
            const reader = new FileReader();
            
            reader.onload = function(e) {
                try {
                    if (file.name.endsWith('.json')) {
                        // 處理JSON檔案
                        const importedData = JSON.parse(e.target.result);
                        
                        // 檢查是否為合併格式的JSON
                        if (Array.isArray(importedData) && importedData.length > 0 && importedData[0].serviceHistory) {
                            // 處理合併格式的JSON
                            importedData.forEach(item => {
                                // 檢查客戶是否已存在
                                const existingCustomerIndex = customers.findIndex(c => c.phone === item.phone);
                                
                                if (existingCustomerIndex === -1) {
                                    // 新增客戶
                                    customers.push({
                                        name: item.name,
                                        phone: item.phone,
                                        totalPoints: item.totalPoints,
                                        usedPoints: item.usedPoints
                                    });
                                    
                                    // 新增服務歷史
                                    if (item.serviceHistory && Array.isArray(item.serviceHistory)) {
                                        serviceHistory = serviceHistory.concat(item.serviceHistory);
                                    }
                                }
                            });
                        } else if (importedData.customers && importedData.serviceHistory) {
                            // 處理分開格式的JSON
                            // 新增客戶
                            importedData.customers.forEach(newCustomer => {
                                const existingCustomerIndex = customers.findIndex(c => c.phone === newCustomer.phone);
                                if (existingCustomerIndex === -1) {
                                    customers.push(newCustomer);
                                }
                            });
                            
                            // 新增服務歷史
                            importedData.serviceHistory.forEach(newHistory => {
                                const existingHistoryIndex = serviceHistory.findIndex(h => 
                                    h.phone === newHistory.phone && 
                                    h.service === newHistory.service && 
                                    h.time === newHistory.time
                                );
                                if (existingHistoryIndex === -1) {
                                    serviceHistory.push(newHistory);
                                }
                            });
                        } else {
                            showNotification('無效的JSON格式', 'error');
                            return;
                        }
                        
                        saveData();
                        displayAllCustomers();
                        showNotification('資料已成功匯入', 'success');
                    } else if (file.name.endsWith('.xlsx')) {
                        // 處理Excel檔案
                        const data = new Uint8Array(e.target.result);
                        const workbook = XLSX.read(data, { type: 'array' });
                        
                        // 假設Excel有一個工作表
                        const sheet = workbook.Sheets[workbook.SheetNames[0]];
                        
                        if (sheet) {
                            const importedData = XLSX.utils.sheet_to_json(sheet, {header: ['name', 'phone', 'usedPoints', 'totalPoints']});
                            
                            // 移除標題行
                            if (importedData.length > 0 && importedData[0].name === '姓名') {
                                importedData.shift();
                            }
                            
                            if (importedData.length > 0) {
                                // 確保資料格式正確
                                const validCustomers = importedData.filter(item => 
                                    item.name && item.phone && 
                                    typeof item.usedPoints !== 'undefined' && 
                                    typeof item.totalPoints !== 'undefined'
                                );
                                
                                if (validCustomers.length > 0) {
                                    // 新增客戶，不覆蓋現有客戶
                                    validCustomers.forEach(newCustomer => {
                                        const existingCustomerIndex = customers.findIndex(c => c.phone === newCustomer.phone);
                                        if (existingCustomerIndex === -1) {
                                            customers.push(newCustomer);
                                        }
                                    });
                                    
                                    saveData();
                                    displayAllCustomers();
                                    showNotification('客戶資料已成功匯入', 'success');
                                } else {
                                    showNotification('Excel檔案中沒有有效的客戶資料', 'error');
                                }
                            } else {
                                showNotification('Excel檔案中沒有客戶資料', 'error');
                            }
                        } else {
                            showNotification('無效的Excel格式', 'error');
                        }
                    } else {
                        showNotification('不支援的檔案格式', 'error');
                    }
                } catch (error) {
                    console.error('匯入資料時發生錯誤:', error);
                    showNotification('匯入資料時發生錯誤', 'error');
                }
            };
            
            if (file.name.endsWith('.json')) {
                reader.readAsText(file);
            } else if (file.name.endsWith('.xlsx')) {
                reader.readAsArrayBuffer(file);
            } else {
                showNotification('不支援的檔案格式', 'error');
            }
        }
        
        // 保存檔案的輔助函數
        function saveAs(blob, fileName) {
            const link = document.createElement('a');
            link.href = URL.createObjectURL(blob);
            link.download = fileName;
            link.click();
            URL.revokeObjectURL(link.href);
        }
